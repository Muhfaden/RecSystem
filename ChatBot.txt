# Define valid options
valid_options = ['Jeddah', 'Riyadh', 'Al Ula']

print("Hi I'm Jawalan.. Where is your destination ? ")
print("")

# Loop until valid option is chosen
while True:
    # Ask user for input
    city = input("Where do you want to go? Please choose from the following options: Jeddah, Riyadh, Al Ula: ")
    
    # Check if input is valid
    if city.title() in valid_options:
        print("Great, you have chosen to go to", city.title())
        break
    else:
        print("Sorry, please choose a valid option.")

# Define valid categories
valid_categories = ['Entertainment', 'Adventure', 'Recreation']

while True:
    # Ask user for input
    category = input("What category do you want to see? Please choose from the following options: Entertainment, Adventure, Recreation: ")
    
    # Check if input is valid
    if category.title() in valid_categories:
        print("These are the suggested places according to your choice:", category.title())
        break
    else:
        print("Sorry, please choose a valid option.")
